43  void Shader::CompileShaders(const char* vertexSource, const char* fragmentSource) {
44  const u32 vertexShader = glCreateShader(GL_VERTEX_SHADER);
45  glShaderSource(vertexShader, 1, &vertexSource, None);
46  glCompileShader(vertexShader);
47  CheckErrors(vertexShader);
48 
49  const u32 fragmentShader = glCreateShader(GL_FRAGMENT_SHADER);
50  glShaderSource(fragmentShader, 1, &fragmentSource, None);
51  glCompileShader(fragmentShader);
52  CheckErrors(fragmentShader);
53 
54  mProgramID = glCreateProgram();

57  glLinkProgram(mProgramID);

60
